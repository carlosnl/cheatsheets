--
-- VALORES REPETIDO DE DOS TABLAS
--
SELECT mail
    FROM
    (
        SELECT mail
        FROM tabla1
        UNION ALL
        SELECT mail
        FROM tabla2
    )  t
    GROUP BY mail
    HAVING COUNT(*) > 1
